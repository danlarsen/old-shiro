namespace SemiCompiled

    {

            Console.Clear();
        }

            return Console.ReadLine();
        }

        [ShiroMethod("emit", 1)]
        public static string MsgBox(string msg)
        {
            Console.WriteLine(msg);
            return msg;
        }
